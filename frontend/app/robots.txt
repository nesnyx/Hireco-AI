import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: '/private/', // Folder yang tidak ingin diindeks (misal dashboard admin)
    },
    sitemap: 'https://hireco.nadinata.org/sitemap.xml',
  }
}
